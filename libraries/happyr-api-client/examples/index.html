This code below is a minor reference about how you can use the library.

[php]
use HappyR\ApiClient\HappyrApi;
use HappyR\ApiClient\Configuration;
use HappyR\ApiClient\Entity\User\User;
use HappyR\ApiClient\Entity\Potential\Pattern;

class myClass
{
private $api=null;

public function getApi()
{
if($this->api==null){
$config=new Configuration('myUsername','myToken');
$this->api=new HappyrApi($config);
}

return $this->api;
}

public function printPatterns()
{
$profilePatterns=$this->getApi()->getPotentialApi()->getPatterns();
foreach($profilePatterns as $pattern){
echo $pattern->name."<br />";
}
}

public function getNextStatement(User $user, Pattern $pattern)
{
$statement=$this->getApi()->getPotentialApi()->getStatement($user, $pattern);
echo $statement->sentence;
}
// ---
}
